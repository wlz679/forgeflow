# i18n Drift Fix — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Eliminate i18n drift by (a) migrating translations from a single `translations.ts` to per-locale JSON files, (b) adding a build-time drift detector that fails the build on missing translations, and (c) repairing the 17 known missing entries across 13 engines.

**Architecture:** Replace `src/i18n/translations.ts` (hardcoded `{en,zh}` object) with two flat JSON files (`src/i18n/locales/{en,zh}.json`) auto-discovered by Vite's `import.meta.glob`. Add `scripts/check-i18n-completeness.mjs` that scans every engine's `inputs[]`/`howToUse[]`/`faq[]` plus the page template's preset blocks, computes expected keys, and fails the build when any are missing from any locale. Tighten `t()` so it never returns a raw key to the page.

**Tech Stack:** Astro 4 + TypeScript, Node.js `node:test` (built-in, zero new deps), `import.meta.glob` (Vite-native), no runtime i18n library.

```mjs
#!/usr/bin/env node
// scripts/check-i18n-completeness.mjs
// Drift detector: compares engine definitions + page template against locale JSON files.
// Exits 1 on missing keys (drift), 0 on clean. --strict also fails on orphan keys.

import fs from 'node:fs';
import path from 'node:path';

const args = process.argv.slice(2);
const strict = args.includes('--strict');
const rootIdx = args.indexOf('--root');
const ROOT = rootIdx >= 0
  ? path.resolve(args[rootIdx + 1])
  : path.resolve(process.cwd());

const ENGINES_DIR = path.join(ROOT, 'src/engines');
const PAGE_TEMPLATE = path.join(ROOT, 'src/pages/[lang]/[slug].astro');
const LOCALES_DIR = path.join(ROOT, 'src/i18n/locales');

// --- Engine parser --------------------------------------------------------

function parseEngines() {
  const files = fs.readdirSync(ENGINES_DIR).filter(f => f.endsWith('.ts'));
  const engines = [];

  for (const file of files) {
    const content = fs.readFileSync(path.join(ENGINES_DIR, file), 'utf8');
    const slug = content.match(/slug:\s*['"]([^'"]+)['"]/)?.[1];
    if (!slug) continue;

    // howToUse array: count top-level quoted strings
    const howToUseMatch = content.match(/howToUse:\s*\[([\s\S]*?)\n\s*\],/);
    const howToUseCount = howToUseMatch
      ? howToUseMatch[1].split('\n').filter(l => /^\s*['"`]/.test(l)).length
      : 0;

    // faq array: count top-level { ... } objects
    const faqMatch = content.match(/faq:\s*\[([\s\S]*?)\n\s*\],/);
    const faqCount = faqMatch
      ? faqMatch[1].split('\n').filter(l => /^\s*\{/.test(l)).length
      : 0;

    // inputs array: match `name: 'foo'` occurrences
    const inputsMatch = content.match(/inputs:\s*\[([\s\S]*?)\n\s*\],/);
    const inputNames = [];
    if (inputsMatch) {
      const nameRe = /name:\s*['"]([^'"]+)['"]/g;
      let m;
      while ((m = nameRe.exec(inputsMatch[1])) !== null) {
        inputNames.push(m[1]);
      }
    }

    engines.push({ slug, howToUseCount, faqCount, inputNames });
  }
  return engines;
}

// --- Page template parser (presets) ---------------------------------------

function parsePresets() {
  if (!fs.existsSync(PAGE_TEMPLATE)) return [];
  const content = fs.readFileSync(PAGE_TEMPLATE, 'utf8');

  // Split by `slug === 'X'` guards
  const slugGuards = content.matchAll(/slug\s*===\s*'([^']+)'/g);
  const presets = [];

  for (const guard of slugGuards) {
    const slug = guard[1];
    // Find the next `{ key: '...' }` literal after this guard
    const afterGuard = content.slice(guard.index + guard[0].length, guard.index + 5000);
    const keyRe = /key:\s*['"]([^'"]+)['"]/g;
    let m;
    const seen = new Set();
    while ((m = keyRe.exec(afterGuard)) !== null) {
      if (seen.has(m[1])) break;  // stop at next slug guard's array
      seen.add(m[1]);
      presets.push({ slug, key: m[1] });
    }
  }
  return presets;
}

// --- Expected key builder -------------------------------------------------

function expectedKeys(engines, presets) {
  const keys = new Set();
  for (const e of engines) {
    keys.add(`tools.${e.slug}.title`);
    keys.add(`tools.${e.slug}.description`);
    for (const inputName of e.inputNames) {
      keys.add(`tools.${e.slug}.input.${inputName}.label`);
      keys.add(`tools.${e.slug}.input.${inputName}.placeholder`);
    }
    for (let i = 0; i < e.howToUseCount; i++) {
      keys.add(`tools.${e.slug}.how_to_use.${i}`);
    }
    for (let i = 0; i < e.faqCount; i++) {
      keys.add(`tools.${e.slug}.faq.${i}.q`);
      keys.add(`tools.${e.slug}.faq.${i}.a`);
    }
  }
  for (const p of presets) {
    keys.add(`tools.${p.slug}.preset.${p.key}`);
  }
  return keys;
}

// --- Locale loader --------------------------------------------------------

function loadLocales() {
  if (!fs.existsSync(LOCALES_DIR)) {
    console.error(`[check:i18n] Locales directory not found: ${LOCALES_DIR}`);
    process.exit(1);
  }
  const locales = {};
  for (const file of fs.readdirSync(LOCALES_DIR).filter(f => f.endsWith('.json'))) {
    const code = file.replace('.json', '');
    locales[code] = JSON.parse(fs.readFileSync(path.join(LOCALES_DIR, file), 'utf8'));
  }
  return locales;
}

// --- Report ---------------------------------------------------------------

function main() {
  const engines = parseEngines();
  const presets = parsePresets();
  const expected = expectedKeys(engines, presets);
  const locales = loadLocales();

  const driftByLocale = {};   // lang -> Set<missing key>
  const orphansByLocale = {}; // lang -> Set<orphan key>

  for (const [lang, dict] of Object.entries(locales)) {
    const present = new Set(Object.keys(dict));
    driftByLocale[lang] = new Set([...expected].filter(k => !present.has(k)));
    orphansByLocale[lang] = new Set([...present].filter(k => !expected.has(k)));
  }

  let driftCount = 0;
  let orphanCount = 0;
  const lines = [];
  lines.push(`Scanning ${engines.length} engines + preset blocks...`);

  // Per-engine summary
  for (const e of engines) {
    const slugMissing = {};
    for (const lang of Object.keys(locales)) {
      const missing = [...driftByLocale[lang]].filter(k => k.startsWith(`tools.${e.slug}.`));
      if (missing.length > 0) {
        slugMissing[lang] = missing;
        driftCount += missing.length;
      }
    }
    if (Object.keys(slugMissing).length === 0) {
      lines.push(`  ✓ ${e.slug}`);
    } else {
      lines.push(`  ❌ ${e.slug}`);
      for (const [lang, missing] of Object.entries(slugMissing)) {
        lines.push(`     ${lang} missing: ${missing.join(', ')}`);
      }
    }
  }

  // Orphan warnings
  for (const [lang, orphans] of Object.entries(orphansByLocale)) {
    if (orphans.size > 0) {
      orphanCount += orphans.size;
      for (const key of orphans) {
        lines.push(`  ⚠ orphan in ${lang}.json: ${key} (no engine references)`);
      }
    }
  }

  lines.push('');
  if (driftCount > 0 || (strict && orphanCount > 0)) {
    const parts = [];
    if (driftCount > 0) parts.push(`${driftCount} missing keys across ${Object.keys(locales).length} locales`);
    if (strict && orphanCount > 0) parts.push(`${orphanCount} orphan keys`);
    lines.push(`─── Drift detected: ${parts.join(', ')}. Build aborted. ───`);
    console.log(lines.join('\n'));
    process.exit(1);
  } else {
    lines.push(`─── Clean. ${engines.length} engines, ${expected.size} keys verified. ───`);
    console.log(lines.join('\n'));
  }
}

main();
```

#!/usr/bin/env node
// scripts/_migrate-translations.mjs
// ONE-SHOT: Reads src/i18n/translations.ts and emits src/i18n/locales/{en,zh}.json
// Delete this file after running.

import fs from 'node:fs';
import path from 'node:path';

const SRC = 'src/i18n/translations.ts';
const OUT_DIR = 'src/i18n/locales';

const content = fs.readFileSync(SRC, 'utf8');
const en = {};
const zh = {};

// Match: 'some.key': { en: '...', zh: '...' },
// Single-quoted strings; values may contain escaped quotes.
const entryRe = /'([^']+)':\s*\{\s*en:\s*'((?:[^'\\]|\\.)*)',\s*zh:\s*'((?:[^'\\]|\\.)*)'\s*,?\s*\}/g;

let m;
let count = 0;
while ((m = entryRe.exec(content)) !== null) {
  const [, key, enVal, zhVal] = m;
  en[key] = unescape(enVal);
  zh[key] = unescape(zhVal);
  count++;
}

fs.mkdirSync(OUT_DIR, { recursive: true });
fs.writeFileSync(path.join(OUT_DIR, 'en.json'), JSON.stringify(en, null, 2) + '\n');
fs.writeFileSync(path.join(OUT_DIR, 'zh.json'), JSON.stringify(zh, null, 2) + '\n');

console.log(`Migrated ${count} entries to ${OUT_DIR}/{en,zh}.json`);
```

import en from './locales/en.json';
import zh from './locales/zh.json';

const locales: Record<string, Record<string, string>> = { en, zh };

export const supportedLangs = Object.keys(locales).sort();
export type Lang = typeof supportedLangs[number];

export type AstroLike = {
  url: URL;
  params?: Record<string, string | undefined>;
  cookies?: { get(name: string): { value: string } | undefined };
};

export function getLang(astro: AstroLike): Lang {
  const p = astro.params?.lang;
  if (p === 'zh' || p === 'en') return p;
  const q = astro.url.searchParams.get('lang');
  if (q === 'zh' || q === 'en') return q;
  const c = astro.cookies?.get('lang');
  if (c && (c.value === 'zh' || c.value === 'en')) return c.value;
  return 'en';
}

function interpolate(text: string, vars?: Record<string, string>): string {
  if (!vars) return text;
  return Object.entries(vars).reduce(
    (acc, [k, v]) => acc.split(`{${k}}`).join(v),
    text
  );
}

export function t(key: string, lang: Lang, vars?: Record<string, string>): string {
  const localized = locales[lang]?.[key];
  if (localized !== undefined) return interpolate(localized, vars);

  const english = locales.en?.[key];
  if (english !== undefined) {
    if (import.meta.env?.DEV) {
      console.warn(`[i18n] Missing ${lang} translation for "${key}", falling back to en`);
    }
    return interpolate(english, vars);
  }

  const msg = `[i18n] Missing translation for "${key}" (no fallback)`;
  if (import.meta.env?.DEV) {
    console.error(msg);
    return `[[${key}]]`;
  }
  console.error(msg);
  return '';
}
```

- [ ] **Step 1: Write the test**

`tests/build/no-raw-keys.test.mjs`:

```mjs
import { test } from 'node:test';
import assert from 'node:assert/strict';
import fs from 'node:fs';
import path from 'node:path';

const DIST = 'dist';

// This test requires `dist/` to exist; it is generated by `pnpm build`.
// In CI, build runs before tests. Locally, run `pnpm build` first.

test('dist/ exists (run pnpm build first)', () => {
  assert.ok(fs.existsSync(DIST), 'dist/ not found — run `pnpm build` first');
});

// Pattern matches raw i18n keys like tools.solopreneur-foo-calculator.how_to_use.5
const RAW_KEY_PATTERN = /tools\.[a-z0-9-]+\.[a-z_.]+\.\d+/g;

function* walkHtml(dir) {
  for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
    const full = path.join(dir, entry.name);
    if (entry.isDirectory()) {
      yield* walkHtml(full);
    } else if (entry.name.endsWith('.html')) {
      yield full;
    }
  }
}

test('no raw i18n keys in dist/en/**/*.html', () => {
  const enDir = path.join(DIST, 'en');
  if (!fs.existsSync(enDir)) return;  // skipped if dist not built
  const offenders = [];
  for (const file of walkHtml(enDir)) {
    const html = fs.readFileSync(file, 'utf8');
    const matches = html.match(RAW_KEY_PATTERN) || [];
    if (matches.length > 0) {
      offenders.push({ file, matches: [...new Set(matches)] });
    }
  }
  assert.equal(offenders.length, 0,
    `Found raw keys in dist/en/:\n${offenders.map(o => `  ${o.file}: ${o.matches.join(', ')}`).join('\n')}`
  );
});

test('no raw i18n keys in dist/zh/**/*.html', () => {
  const zhDir = path.join(DIST, 'zh');
  if (!fs.existsSync(zhDir)) return;
  const offenders = [];
  for (const file of walkHtml(zhDir)) {
    const html = fs.readFileSync(file, 'utf8');
    const matches = html.match(RAW_KEY_PATTERN) || [];
    if (matches.length > 0) {
      offenders.push({ file, matches: [...new Set(matches)] });
    }
  }
  assert.equal(offenders.length, 0,
    `Found raw keys in dist/zh/:\n${offenders.map(o => `  ${o.file}: ${o.matches.join(', ')}`).join('\n')}`
  );
});
```
